C Code For LabWork #5



#include 

struct Books { 
    char name[50];
    char author[50];
    int nop;


};

int main() {

    struct Books book[3];
    for (int i=0; i<3; i++) {
        printf("Enter details for Book %d:\n", i+1);
        printf("Book Name: ");
        scanf("%s", &book[i].name);
        printf("Number of pages: ");
        scanf("%d", &book[i].nop);
        printf("Author: ");
        scanf("%s", &book[i].author);
    }
    printf("\nStudent Details:\n");
    for (int i = 0; i < 3; i++) {
        printf("Book Name: %s, Number of pages: %d, Author: %s\n", book[i].name, book[i].nop, book[i].author);
    }

    return 0;

}
