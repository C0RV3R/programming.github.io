package mazesolver_datastructeres;

import java.io.File;
import java.util.Scanner;
import java.io.IOException; 


public class MazeSolver_ZerosOnes {

    static final int size = 15;   
    static int[][] maze = new int[size][size];
    static boolean[][] isVisited = new boolean[size][size];

    static class Place {
        int row, col;
        
        Place(int rowVal, int colVal) {
            row = rowVal;
            col = colVal;
        }
    }

    static class Stack {
        private int top;
        private Place[] data;

        public Stack(int capacity) {
            data = new Place[capacity];
            top = -1;
        }

        public void push(Place item) {
            data[++top] = item;
        }

        public Place pop() {
            if (isEmpty()) return null;
            return data[top--];
        }

        public boolean isEmpty() {
            return top == -1;
        }
    }

    public static void main(String[] args) throws IOException {
    	
    	System.out.println("Enterance -> Row: 1 Col: 2");

        Scanner scanner = new Scanner(new File("maze.txt"));
        int row = 0;

        while (row < size) {
            String line = scanner.nextLine();
            for (int col = 0; col < size && col < line.length(); col++) {
                maze[row][col] = line.charAt(col) - '0';
            }
            row++;
        }
        scanner.close();

        findThePath();
    }

    static void findThePath() {
        for (int i = 0; i < size; i++) {
            for (int a = 0; a < size; a++) {
                isVisited[i][a] = false;
            }
        }
        
        int startCol = 1; // Hard-Coded Start ._. (0, 1) -> (Row 1, Col 2)
        
        Stack SearchStack = new Stack(size * size);
        SearchStack.push(new Place(0, startCol)); 
        isVisited[0][startCol] = true;

        int[] directionRow = {-1, 1, 0, 0}; 
        int[] directionCol = {0, 0, -1, 1};

        while (!SearchStack.isEmpty()) {
            Place current = SearchStack.pop();

            if (current.row == size - 1) {
                System.out.println("Finish -> Row: " + (current.row + 1) + " Col: " + (current.col + 1));
                return;
            }

            for (int i = 0; i <= 3; i++) {
                int nextRow = current.row + directionRow[i];
                int nextCol = current.col + directionCol[i];

                if (tryToMove(nextRow, nextCol)) {
                    isVisited[nextRow][nextCol] = true;
                    SearchStack.push(new Place(nextRow, nextCol));
                    System.out.println("Row: " + (nextRow + 1) + " Col: " + (nextCol + 1));
                }
            }
        }

        System.out.println("Oops! There is no path.");
    }

    static boolean tryToMove(int nextRow, int nextCol) {
        if (nextRow < 0 || nextCol < 0 || nextRow >= size || nextCol >= size)
            return false;
        if (maze[nextRow][nextCol] != 0)
            return false;
        if (isVisited[nextRow][nextCol])
            return false;
        return true;
    }
}
